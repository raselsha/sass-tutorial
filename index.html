Sass tutorial

  1. Install node js
  2. Install visual studio code
  3. VS terminal type npm init
  4. VS terminal type npm install bootstrap
  5. VS terminal type npm install gulp-sass sass --save-dev
  6. create a file 'gulpfile.js'

inside gulpfile.js 

            
                const {src,dest,watch,series} = require('gulp');
                const sass = require('gulp-sass')(require('sass'));

                function buildStyle(){
                    return src('sass/*.scss')
                        .pipe(sass())
                        .pipe(dest('css'))
                }

                function watchTask(){
                    watch(['sass/*.scss'],buildStyle);
                }

                exports.default = series(buildStyle,watchTask);
